### Algorithm 1: SM-2 (SuperMemo 2) ‚Äî Default

```swift
class SM2Algorithm: SpacedRepetitionAlgorithm {
    let name = "SM-2"
    let description = "–ö–ª–∞—Å—Å–∏—á–µ—Å–∫–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º SuperMemo. –ê–¥–∞–ø—Ç–∏—Ä—É–µ—Ç—Å—è –ø–æ–¥ –≤–∞—à—É –ø–∞–º—è—Ç—å."

    func calculateNextReview(for card: Card, quality: ReviewQuality) -> CardSchedule {
        var easeFactor = card.easeFactor ?? 2.5
        var interval = card.interval ?? 0
        var repetitions = card.repetitions ?? 0

        // Quality: 0 = forgot, 5 = perfect
        let q: Double = quality == .remembered ? 4.0 : 1.0

        // Update ease factor
        easeFactor = max(1.3, easeFactor + 0.1 - (5 - q) * (0.08 + (5 - q) * 0.02))

        if quality == .forgot {
            // Reset
            repetitions = 0
            interval = 1
        } else {
            // Success
            repetitions += 1

            switch repetitions {
            case 1:
                interval = 1
            case 2:
                interval = 6
            default:
                interval = Int(Double(interval) * easeFactor)
            }
        }

        let nextDate = Calendar.current.date(byAdding: .day, value: interval, to: Date())!

        let status: CardStatus
        if repetitions == 0 {
            status = .learning
        } else if interval >= 21 {
            status = .mastered
        } else {
            status = .review
        }

        return CardSchedule(
            nextReviewDate: nextDate,
            easeFactor: easeFactor,
            interval: interval,
            status: status
        )
    }
}
```

### Algorithm 2: Leitner System

```swift
class LeitnerAlgorithm: SpacedRepetitionAlgorithm {
    let name = "Leitner"
    let description = "–°–∏—Å—Ç–µ–º–∞ 5 –∫–æ—Ä–æ–±–æ–∫. –ü—Ä–æ—Å—Ç–∞—è –∏ –ø–æ–Ω—è—Ç–Ω–∞—è –¥–ª—è –Ω–æ–≤–∏—á–∫–æ–≤."

    // Box intervals in days
    private let boxIntervals = [1, 2, 4, 7, 14]

    func calculateNextReview(for card: Card, quality: ReviewQuality) -> CardSchedule {
        var box = card.leitnerBox ?? 0 // 0-4

        if quality == .remembered {
            // Move to next box
            box = min(box + 1, 4)
        } else {
            // Back to box 0
            box = 0
        }

        let interval = boxIntervals[box]
        let nextDate = Calendar.current.date(byAdding: .day, value: interval, to: Date())!

        let status: CardStatus
        switch box {
        case 0:
            status = .learning
        case 1, 2:
            status = .review
        default:
            status = .mastered
        }

        return CardSchedule(
            nextReviewDate: nextDate,
            easeFactor: 2.5, // Not used in Leitner
            interval: interval,
            status: status
        )
    }
}
```

class SimpleAlgorithm: SpacedRepetitionAlgorithm {
    let name = "Simple"
    let description = "–§–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –∏–Ω—Ç–µ—Ä–≤–∞–ª—ã: 1-3-7-14-30 –¥–Ω–µ–π. –ü—Ä–µ–¥—Å–∫–∞–∑—É–µ–º–æ."

    private let intervals = [1, 3, 7, 14, 30]

    func calculateNextReview(for card: Card, quality: ReviewQuality) -> CardSchedule {
        var level = card.simpleLevel ?? 0 // 0-4

        if quality == .remembered {
            level = min(level + 1, 4)
        } else {
            level = max(level - 1, 0)
        }

        let interval = intervals[level]
        let nextDate = Calendar.current.date(byAdding: .day, value: interval, to: Date())!

        let status: CardStatus
        switch level {
        case 0:
            status = .learning
        case 1, 2:
            status = .review
        default:
            status = .mastered
        }

        return CardSchedule(
            nextReviewDate: nextDate,
            easeFactor: 2.5,
            interval: interval,
            status: status
        )
    }
}

class GoogleSheetsImportService {
    func importFromPublicURL(_ urlString: String) async throws -> [Card] {
        // Convert to CSV export URL
        // Example: https://docs.google.com/spreadsheets/d/SPREADSHEET_ID/edit
        // ‚Üí https://docs.google.com/spreadsheets/d/SPREADSHEET_ID/export?format=csv

        guard let spreadsheetId = extractSpreadsheetId(from: urlString) else {
            throw ImportError.invalidURL
        }

        let csvURL = URL(string: "https://docs.google.com/spreadsheets/d/\(spreadsheetId)/export?format=csv")!

        let (data, _) = try await URLSession.shared.data(from: csvURL)
        let content = String(data: data, encoding: .utf8) ?? ""

        // Parse CSV
        return try CSVImportService().parseCSV(content)
    }

    private func extractSpreadsheetId(from urlString: String) -> String? {
        // Extract ID from URL
        let pattern = #"/d/([a-zA-Z0-9-_]+)"#
        guard let regex = try? NSRegularExpression(pattern: pattern),
              let match = regex.firstMatch(in: urlString, range: NSRange(urlString.startIndex..., in: urlString)),
              let range = Range(match.range(at: 1), in: urlString) else {
            return nil
        }
        return String(urlString[range])
    }
}
